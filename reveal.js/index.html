call and apply

var Cat = function(name) {
    this.name = name;
};
Cat.prototype.meow = function(food) {
    alert(this.name + " " + food);
}
var cat = new Cat();

var obj = { name : "tiger" };
cat.meow.call(obj, "food1");
cat.meow.apply(obj, ["food1"]);
